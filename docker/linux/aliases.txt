
# -- Dockerfile にて追加、マウント済みのためコンテナ内でも編集可 --
# cron の起動
alias startcron='/etc/init.d/cron start'
# cron の停止
alias stopcron='/etc/init.d/cron stop'
# cron の再起動
alias restartcron='/etc/init.d/cron restart'
# cron が動いてるか確認
alias cronstatus='/etc/init.d/cron status'
# systemd 起動時からのログすべてを less コマンドで表示
alias cronlogall='less /var/log/cron.log'
# cron のログを tail 表示
alias cronlogs='tail /var/log/cron.log'
# cron.log の記録を開始する
alias cronloginit='/etc/init.d/rsyslog restart'
